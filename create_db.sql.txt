-- 3. Создание схемы
CREATE SCHEMA IF NOT EXISTS app AUTHORIZATION app;

-- 4. Установка схемы по умолчанию
SET search_path TO app;

-- 5. Таблица типов партнеров
CREATE TABLE IF NOT EXISTS app.partner_types (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 6. Таблица партнеров
CREATE TABLE IF NOT EXISTS app.partners (
    id SERIAL PRIMARY KEY,
    type_id INTEGER NOT NULL,
    company_name VARCHAR(250) NOT NULL,
    legal_address TEXT,
    inn_number VARCHAR(12) UNIQUE,
    director_fullname VARCHAR(200),
    contact_phone VARCHAR(20),
    contact_email VARCHAR(100),
    partner_rating INTEGER DEFAULT 0 CHECK (partner_rating >= 0),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_partners_type_id FOREIGN KEY (type_id) 
        REFERENCES app.partner_types(id) ON DELETE RESTRICT
);

-- 7. Таблица продукции
CREATE TABLE IF NOT EXISTS app.products (
    id SERIAL PRIMARY KEY,
    product_article VARCHAR(50) UNIQUE NOT NULL,
    product_type VARCHAR(100),
    product_name VARCHAR(250) NOT NULL,
    minimal_price DECIMAL(12, 2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 8. Таблица истории продаж
CREATE TABLE IF NOT EXISTS app.sale_history (
    id SERIAL PRIMARY KEY,
    partner_id INTEGER NOT NULL,
    product_id INTEGER NOT NULL,
    quantity_sold INTEGER NOT NULL CHECK (quantity_sold > 0),
    sale_date DATE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_sale_history_partner_id FOREIGN KEY (partner_id) 
        REFERENCES app.partners(id) ON DELETE CASCADE,
    CONSTRAINT fk_sale_history_product_id FOREIGN KEY (product_id) 
        REFERENCES app.products(id) ON DELETE RESTRICT
);

-- 9. Индексы
CREATE INDEX IF NOT EXISTS idx_partners_type_id ON app.partners(type_id);
CREATE INDEX IF NOT EXISTS idx_partners_company_name ON app.partners(company_name);
CREATE INDEX IF NOT EXISTS idx_sale_history_partner_id ON app.sale_history(partner_id);
CREATE INDEX IF NOT EXISTS idx_sale_history_sale_date ON app.sale_history(sale_date);
CREATE INDEX IF NOT EXISTS idx_sale_history_partner_date ON app.sale_history(partner_id, sale_date);
CREATE INDEX IF NOT EXISTS idx_products_article ON app.products(product_article);
CREATE INDEX IF NOT EXISTS idx_products_name ON app.products(product_name);

-- 10. Права доступа
GRANT USAGE ON SCHEMA app TO app;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA app TO app;
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA app TO app;

-- 11. Функция расчета общей суммы продаж партнера
CREATE OR REPLACE FUNCTION app.get_partner_total_sales(p_partner_id INTEGER)
RETURNS DECIMAL(12,2) AS $$
DECLARE
    v_total DECIMAL(12,2);
BEGIN
    SELECT COALESCE(SUM(s.quantity_sold * p.minimal_price), 0)
    INTO v_total
    FROM app.sale_history s
    JOIN app.products p ON s.product_id = p.id
    WHERE s.partner_id = p_partner_id;
    
    RETURN v_total;
END;
$$ LANGUAGE plpgsql;

-- 12. Функция расчета скидки партнера
CREATE OR REPLACE FUNCTION app.calculate_partner_discount(p_partner_id INTEGER)
RETURNS INTEGER AS $$
DECLARE
    v_total_sales DECIMAL(12,2);
    v_discount INTEGER;
BEGIN
    v_total_sales := app.get_partner_total_sales(p_partner_id);
    
    IF v_total_sales > 300000 THEN
        v_discount := 15;
    ELSIF v_total_sales > 50000 THEN
        v_discount := 10;
    ELSIF v_total_sales > 10000 THEN
        v_discount := 5;
    ELSE
        v_discount := 0;
    END IF;
    
    RETURN v_discount;
END;
$$ LANGUAGE plpgsql;

-- 13. Процедура добавления продажи
CREATE OR REPLACE PROCEDURE app.add_sale(
    p_partner_id INTEGER,
    p_product_id INTEGER,
    p_quantity INTEGER,
    p_sale_date DATE DEFAULT CURRENT_DATE
)
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO app.sale_history (partner_id, product_id, quantity_sold, sale_date, created_at)
    VALUES (p_partner_id, p_product_id, p_quantity, p_sale_date, CURRENT_TIMESTAMP);
END;
$$;

-- 14. Функция для триггера обновления updated_at
CREATE OR REPLACE FUNCTION app.update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 15. Триггер для автоматического обновления updated_at
DROP TRIGGER IF EXISTS trigger_partners_updated_at ON app.partners;
CREATE TRIGGER trigger_partners_updated_at
    BEFORE UPDATE ON app.partners
    FOR EACH ROW
    EXECUTE FUNCTION app.update_updated_at_column();
